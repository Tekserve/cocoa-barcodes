#defines


CBC_ASCENDER_TOP

#define CBC_ASCENDER_TOP (1.9 + CBC_TRACK_TOP)

Defines the highest point of the ascender (in millimeters).

CBC_BAR_WIDTH

#define CBC_BAR_WIDTH (0.5)

Width, in millimeters, of a CBC bar. Specifications allow a range: 0.38-0.63mm.

CBC_DESCENDER_BOTTOM

#define CBC_DESCENDER_BOTTOM (0.0)

Defines the lowest point of the descender (in millimeters).

CBC_INCHES_PER_MILLIMETER

#define CBC_INCHES_PER_MILLIMETER (1./25.4)

Number of inches per millimeter (1/25.4).

CBC_TRACK_BOTTOM

#define CBC_TRACK_BOTTOM (1.9 + CBC_DESCENDER_BOTTOM)

Defines the lower boundary of the track (in millimeters).

CBC_TRACK_TOP

#define CBC_TRACK_TOP (1.25 + CBC_TRACK_BOTTOM)
